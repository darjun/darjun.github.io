  "time"

  "github.com/uniplaces/carbon"

func main() {
  t, err := carbon.Create(2020, 02, 11, 0, 0, 0, 0, "Asia/Shanghai")

    log.Fatal(err)

  t.SetWeekEndsAt(time.Saturday)

}
