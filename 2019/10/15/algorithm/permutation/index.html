func getLargestMovableIndex(nums []int, dir []bool) int {
    maxI := -1
    l := len(nums)
    for i, num := range nums {
        if dir[i] {
            if i > 0 && num > nums[i-1] {
                if maxI == -1 || num > nums[maxI] {
                    maxI = i
                }
            }
        } else {
            if i < l-1 && num > nums[i+1] {
                if maxI == -1 || num > nums[maxI] {
                    maxI = i
                }
            }
        }
    }

    return maxI
}
